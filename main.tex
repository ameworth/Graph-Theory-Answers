\chapter{Basic Counting: Sets and Tuples}

\begin{question}
    How large is $[6]\choose{3}$?
\end{question}
\begin{answer}
  From Lemma 1.2(iv) we have that $|\binom{[6]}{3}| = \binom{6}{3} = 20$
\end{answer}

\begin{question} 
  For $A \subseteq \mathbb{N}$ define $S(A) = \Sigma_{a \in A}a$. For how many sets $A \in \binom{[6]}{3}$ is $S(A)$ odd? 
\end{question}
\begin{answer}
$\binom{[6]}{3} = \{ A \subseteq [6] : |A| = 3\}$. In order for $S(A)$ to be odd, we need an text odd number of odd elements (since odd + odd + even = even), and at least one odd number (as even + even + even = even). This then gives 2 possible configurations of sets: 
  \begin{align*}
    \{odd, odd, odd\} \text{ or } \{odd, even, even\} 
  \end{align*} 
  There are $\binom{3}{3}=1$ of the first type, and $\binom{3}{1} \binom{3}{2}= 3 \cdot 3 = 9$ of the second type, thus giving 10 total sets where $S(A)$ is odd. 
\end{answer}

\begin{question}
  Let $n \in \mathbb{N}$. How large are the following family of sets: 
  \begin{align*}
    \mathcal{F}_1 = \{ A \subseteq     &[n] : 1 \in A\} \\
    \mathcal{F}_3 = \{ A \subseteq [n] &: |A \cap [3]| \geq 2\} 
  \end{align*}
\end{question}
\begin{answer}
  For the first part, a quick observation will lead to conjecturing $|\mathcal{F}_1| = 2^{n-1}$\\
  To prove this, proceed by induction on $n \in \mathbb{N}$. The base cases is obviously true for $n=1$, so assume true for $n=k$ and consider $\mathcal{F}_1$ when $n=k+1$. Observe
  \begin{align*}
    \mathcal{F}_1 &= \{ A \subseteq [n] : 1 \in A \} \\
                 &= \{ A \subseteq [k] \cup \{k+1\} : 1 \in A \}
  \end{align*}
  From the inductive hypothesis, we already know that there are $2^{k-1}$ sets containing 1. By appending a the number $k+1$ on the end of all of these sets, we get $2^{k-1}$ additional distinct sets containg 1, and no more since order is irrelevant in sets. Hence $|\mathcal{F}_1| = 2^{k-1} + 2^{k-1} = 2*2^{k-1} = 2^{(k+1)-1}$. \\

  For the second part, we can count how many elements of $[3]$ we get in $A$. Either we get exactly two, or all 3. If you fix which elements are in $A$, given we take two or three, the remaining $n-3$ elements of $\{4, \dots, m\}$ are free, giving $2^{n-3}$ choices each time. This gives us: 
  \begin{align*}
    |\mathcal{F_3}| = (\binom{3}{2} + \binom{3}{3})2^{n-3} = 4 \cdot 2^{n-3} = 2^{n-1}
  \end{align*}
\end{answer}

\begin{question}
  How many sets in $\binom{[6]}{3}$ do not contain two consecutive integers?
\end{question}
\begin{answer}
  There's only 20 sets, so its actually quite easy to just list them. Starting with 1, as the first entry, there must be a gap of at least two between each element of the set, giving us $\{1, 3, 5,\}, \{1, 3, 6,\}, \{1, 4, 6\}$. Now starting with 2, you can only get  $\{2, 4, 6\}$. You cannot start with 3 since your last element would have to be at least 7, which is not in the set. Hence there are 4 such sets. 
\end{answer}

\begin{question}
  Let $n \in \mathbb{N}$ be odd. Find a simple expression for the sum
  \begin{align*}
    \sum\limits_{k>n/2}\binom{n}{k}
  \end{align*}
\end{question}
\begin{answer}
  We have from the binomial theorem that $(1+x)^n = \sum\limits_{k=0}^{n}\binom{n}{k}x^k$ hence the sum of binomial coefficients is $2^n$. Binomial coefficients are symmetric and there is no central term since n is odd (see: Pascal's Triangle), hence: 
  \begin{align*}
    \sum\limits_{k>n/2}\binom{n}{k} = 1/2 \sum\limits_{k=0}^{n}\binom{n}{k} = 2^{n-1}
  \end{align*} 
\end{answer}

\begin{question}
  Let $n \in \mathbb{N}$ for which $k \in \mathbb{N}$ is $\binom{n}{k}$ maximal? Is $k$ unique? 
\end{question}
\begin{answer}
  We have (with a bit of work) that $\binom{n}{k} - \binom{n}{k-1} = \frac{n+1-2k}{n+1}\binom{n+1}{k}$. So for fixed n, we have that the sequence of binomials is increasing when $2k \leq n+1$ and decreasing when $2k \geq n+1$. Hence its maximum is attained when $k = \frac{n+1}{2}$. Since $n$ is a natural number, this is only unique when $n$ is odd. When $n$ is even you can either take the floor or the ceiling.
\end{answer}

\begin{question}
Prove that any graph of order $n \geq 2$ has two vertices of the same degree.    
\end{question}
\begin{answer}
  The maximum degree a vertex can have in any simple graph is $n-1$. Observe the degree sequence: 
  \begin{align*}
    0 \leq d(x_1) \leq d(x_2) \leq \dots \leq d(x_n) \leq n-1 
  \end{align*}
  Since there are $n$ vertices, and only $n-1$ values the degree that any vertex can take, by the pigeonhole principle, two vertices must share the same degree.
\end{answer}

\begin{question}
  Let $G$ be a graph of order $n \geq 2$ in which every vertex has degree at least $(n-1)/2$. Prove that for every pair of distinct non-adjacent vertices $u, v \in V(G)$ there is a third vertex $w$ such that $uw, vw \in E(G)$
\end{question}
\begin{answer}
  Consider the neighborhoods of both vertices $u$ and $v$. By assumption, 
  \begin{align*}
    |\Gamma(u)| \geq \frac{n-1}{2}, \hspace{0.5cm} |\Gamma(v)| \geq \frac{n-1}{2}
  \end{align*}
  and since $u$ and $v$ are non adjacent and we are working with simple graphs, we have that 
  \begin{align*}
    \Gamma(u), \Gamma{v} \subseteq V(G) \setminus \{u,v\}
  \end{align*}
  which has $n-2$ vertices. $|\Gamma(u)| + |\Gamma(v)| \geq \frac{n-1}{2} + \frac{n-1}{2} = n-1$. So by pigeonhole, $\Gamma(u)$ and $\Gamma(v)$ must share a vertex. Label that vertex $w$ and the statement is proved. 
\end{answer}

\begin{question}
  Let $n \in \mathbb{N}$. What are the sizes of the graphs: $C_n, \, K_n, \, K_{n,2n}, K_{n,n}, \, Q_n$?
\end{question}
\begin{answer}
  All of these can be worked out from the handshake lemma: divide the sum of vertex degree by 2.\\
  $|C_n| = n$ as there are $n$ vertices of degree 2. \\
  $|K_n| = \frac{n(n-1)}{2}$ as there are $n$ vertices of degree $n-1$ \\
  $|K_{n,2n}| = 2n^2$ as there are $n$ vertices of degree $2n$, and $2n$ vertices of degree $n$. \\
  $|K_{n,n}| = n^2$ as there are $n$ vertices of degree $n$ and then another $n$ vertices of degree $n$. 
  $|Q_n| = \frac{2^n n}{2}$ as there are $2^n$ vertices of degree $n$.
\end{answer}

\begin{question}
  Let $1 \leq a \leq b$ be integers. If $K_{a,b}$ has order $n$ what is the smallest and largest that the size of $K_{a,b}$ could be?
\end{question}
\begin{answer}
  Obviously, the smallest is when $a=b=1$. For the largest, you can solve via calculus, but intuitively, since $a + b = n$ and from before $|K_{a,b}| = ab$, everytime we increase the value of $a$, $b$ must get smaller (and visa versa). Hence the product is maximised when a and b are balanced: $a=b= n/2$. So the largest size is $n^2/4$.
\end{answer}

\begin{question}
  For which $a, b, t \in \mathbb{N}$ does $K_{a,b}$ contain a copy of $C_t$?  
\end{question}
\begin{answer}
  I opt for a less formal answer in this case. Assume $a,b,t > 1$ since for $a=b=1$ you can only get $C_1$ and $C_1$ is present for all $a,b > 0$.
  Consider the parity of $t$. When $t$ is odd, we would need that two vertices of partition are adjacent, so $t$ must be even. A cycle needs to alternate between the two sides, which means it must use the same number of vertices from each side. If a cycle is of length $t$, it therefore uses $t/2$ vertices from $A$ and $t/2$ vertices from $B$, so we need enough vertices on each side to supply $t/2$ distinct ones. Hence we get $t/2 \leq min\{a,b\} \implies t \leq 2min\{a,b\}$. 
\end{answer}

\begin{question}

\end{question}
\begin{answer}
  For every vertex in partition A, we just pick 2 vertices in partition B, hence $a \cdot \binom{b}{2}$
\end{answer}

\begin{question}
  Suppose $G$ has girth at least 7 and minimum degree at least 50, prove that G must have order at least 122551.
\end{question}
\begin{answer}
  Consider a vertex $v \in V(G)$. By assumption, this vertex has at least 50 neighbours. Consider an arbitrary neighbour $w$, by assumption, $w$ must have at least 50 vertices. This is true for all neighbours of $v$, since if two $w$'s shared a vertex we would have a cycle of length 4. This argument then applies again to 50 vertices adjacent to $w$, implying that we have at least $50^3 = 125000$ vertices.
\end{answer}

\begin{question}
  Show that if $T$ contains a vertex of degree $d$, then it must contain at least $d$ leaves. Is this tight?
\end{question}
\begin{answer}
  Let $T$ be a tree and let $v \in V$ have degree $d$, let the neighbours of $v$ be $u_1, \dots, u_d$. Removing $v$ and its edges splits the graph into $d$ connected components $C_1, \dots, C_d$, each being a tree. Every finite tree has at least one leaf, choose a leaf $w_i \in C_i$. If $C_i$ is the single vertex $u_i$, then $u_i = w_i$ is a leaf of $T$. If $C_i$ has more than one vertex, then its leaf $w_i$ is not $u_i$, so $w_i$ has degree 1 inside $C_i$ and is not adjacent to $v$. Therefore $w_i$ has degree 1 in $T$, and thus is a leaf in $T$. $w_1, \dots, w_d$ lie in different components $C_i$. So T has at least $d$ leaves. 
  Tight for $K_{1,d}$. 
\end{answer}

\section{Euler Circuits}

\begin{question}
  Let $G$ be a connected graph in which all vertices have even degree except two which have odd degree. Show that there is a walk in $G$ using every edge exactly once. 
\end{question}
\begin{answer}
  Let the two vertices of odd degree be $u$, $v$. Create a new graph $G'$ by adding a new edge between $u$ and $v$ (if they're adjacent, just add a parallel edge). Then $G'$ is a connected graph wherein all vertices have even degree, and hence has an eulerian circuit $C$. If we remove the edge $uv$ from $C$, what remains is an eulerian trail whose endpoints are $u$ and $v$. 
\end{answer}

\section{Bipartite Graphs}

\begin{question}
  Let $G = (V, E)$ be a bipartite graph with non-empty vertex classes $X$ and $Y$. Show that if every vertex in $X$ has degree $x \geq 1$ and every vertex in $Y$ has degree $y \geq 1$, then $x|X| = y|Y|$.
\end{question}
\begin{answer}
  This follows from a short double counting argument. Every edge of $G$ has one endpoint in $X$ and one endpoint in $Y$, so we have: 
  \begin{align*}
    \sum\limits_{v \in X} deg(v)  = |E| = \sum\limits_{v \in Y} deg(v)
  \end{align*}
  since $deg(v) = x$ for $x \in X$ and $deg(v) = y$ for $y \in Y$, the result follows immediately. 
\end{answer}

\begin{question}
  Which of the following graphs are bipartite: $C_n, K_n, E_n, P_n, Q_n$?
\end{question}
\begin{answer}
  Let $n > 1$ since when $n=1$ all graphs are bipartite.  
  $C_n$: Since a graph is bipartite iff it contains no odd cycle, $C_n$ is only bipartite when $n$ is even. \\
  $K_n$: only bipartite when $n=2$ as when $n\geq 3$, $K_n$ contains a copy of $C_3$.\\
  $E_n$: for all $n$, since we can just take any partition. \\
  $Q_n$: There are many ways to do this, the simplest way is to notice that we can partition the vertex set into one set containing all vertices with an even number of 1s, and another set containing all vertices with an odd number of 1s. There are no edges within these sets, as two vertices within the each set must differ by at least two coordinates. So $Q_n$ is bipartite for all $n$.
\end{answer}


\section{Graph Colouring}
\begin{question}
  For a graph $G$ let 
  \begin{align*}
    \omega(G) = max\{t : G \text{ contains a copy of } K_t \} 
  \end{align*}
  and 
  \begin{align*}
    \alpha(G) = max\{|A| : A \subseteq V(G) \text{ is an independent set } \} 
  \end{align*}
  Show that $\chi(G) \geq max\{\omega(G), n/\alpha(G)\}$
\end{question}
\begin{answer}
  First prove $\chi(G) \geq \omega(G)$. If $K_t \subset G$ where $\omega(G)=t$ then $\chi(G) \geq \chi(K_t) \geq t = \omega(G)$. \\
  Now prove $\chi(G) \geq n/(\alpha(G))$. Note that a colouring seperates $V(G)$ into colour classes, each of which cannot be bigger than $\alpha(G)$. So we have $n = |V(G)| \leq \chi(G) \cdot \alpha(G) \implies \chi(G) \geq n/(\alpha(G))$. Result follows immediately. 
\end{answer}

\begin{question}
  Let $G$ be a graph with chromatic number $k$. Show that $G$ must have at least $\binom{k}{2}$ edges. 
\end{question}
\begin{answer}
  Chromatic number $k$ means that $V = V_1 \dots{\cup} V_2 \dots{\cup} \dots V_k$ is the smallest partition we can get (i.e. we couldn't get $V = V_1 \dots{\cup} V_2 \dots{\cup} \dots V_{k-1}$, since then it would be $k-1$-colourable). In order to be $k$-partite, we need that each partition has an edge going to every other partition, otherwise two partitions have no edges between them and thus can be collapsed into the same partition. Hence this means we have at least $\binom{k}{2}$ edges. 
\end{answer}

\begin{question}
  Let $G = (A,B;E)$ be the bipartite graph with vertex set $A \dot{\cup} B$ where $A = \{ a_1, \dots, a_n\}$ and $B = \{ b_1, \dots, b_n\}$ and edges: 
  \begin{align*}
    E = \{a_i b_j : 1 \leq i, j \leq n, i \neq j\}  
  \end{align*}
  \begin{enumerate}
    \item Determine $\chi(G)$ and find an ordering of the vertices so that the greedy algorithm uses exactly $\chi(G)$ colours. \\
    \item Find an ordering of the vertices so that the greedy algorithm uses as many colours as possible.
  \end{enumerate}
\end{question}
\begin{answer}
  Since $G$ is a bipartite graph with size greater than 1, we have that $\chi(G) = 2$. The ordering in which the greedy algorithm gives two colours is $V(G) = \{a_1, a_2, \dots, a_n, b_1, \dots, b_n\}$. For $a_1$ to $a_n$, the algorithm gives them all the colour 0, since none are adjacent to eachother. Then from $b_1$ to $b_n$ the algorithm gives all of these vertices the colour 1, since each vertex is adjacent only to vertices with colour 0.\\
  To get the maximal colouring, we arrange the vertices like $\{a_1, b_1, a_2, b_2, \dots, a_n, b_n\}$. This gives us $n$ colours: $a_1$ and $b_1$ are given the colour 0 since they are non adjacent, then $a_2$ is given the colour 1 since it is adjacent to $b_1$. $b_2$ is also given colour 1 since it is adjacent to $a_1$ but not $a_2$. Repeating this process gives colour classes of size two consiting of $\{a_i, b_i\}$ and thus uses $n$ colours. We know that this is maximum since $\Delta(G) + 1 = n-1 + 1 = n$.  
\end{answer}

\chapter{The Probabilistic Method}
\section{Basics}

\begin{question}
  Let $G = (V,E)$ be a graph with $e$ edges. Prove that if $k \geq 2$ then there exists a partition of $V$ into $k$ classes: $V = V_1 \dot{\cup} V_2 \dot{\cup} \dots \dot{\cup} V_k$ such that at least $\lceil (k-1)e/k\rceil$ of the edges in $E$ go between different classes. 
\end{question}
\begin{answer}
  This is very similar to the construction in Prop 4.5. Consider a random partition $V_1 \dot{\cup} V_2 \dot{\cup} \dots \dot{\cup} V_k$ of $V$ as follows. For each $v \in V(G)$ toss a fair $k$-sided dice (i.e. outcomes $1 \leq i \leq k$), placing $v \in V_i$ depending on the value of the dice. Evidently all dice throws are independent.\\
  For an edge $uv \in E(G)$ let $X_{uv}$ be such that 
  \begin{align*}
    X_{uv} =
    \begin{cases}
    1, & \text{if } u, v \text{ are in different parts},\\[4pt]
    0, & \text{otherwise}.
    \end{cases}
  \end{align*}
\end{answer}

\section{Random Graphs}
\begin{question}
  What is the expected number of edges in a random graph $H \in \mathcal{G}(n,p)$?
\end{question}
\begin{answer}
  Let $X_{uv}$ be the indicator variable
  \begin{align*}
    X_{uv} = \begin{cases} 
      1, &\text{if } uv \in E(H)\\
      0, &\text{otherwise} 
    \end{cases}
  \end{align*}
  Each edge is in $E(H)$ with probability $p$, so $\mathbb{E}[X_{uv}] = 1 \cdot \mathbb{P}[uv \in E(H)] = p$. There are $\binom{n}{2}$ potential edges; let X be the total number of edges, then by linearity of expectation: 
  \begin{align*}
    \mathbb{E}[X] = \mathbb{E}[\sum\limits_{uv \in E(H)}X_{uv}] = \sum\limits_{uv \in E(H)}\mathbb{E}[X_{uv}] = \binom{n}{2}p
  \end{align*}
\end{answer}

\begin{question}
  For which $p \in [0,1]$ is it true that $\mathcal{G}(n,p)$ is simply the space of all labeled graphs on vertex set $[n]$ with the uniform probability measure? (i.e. when is generating a graph from $\mathcal{G}(n,p)$ is the same as sampling a graph uniformally at random from the set of all graphs on the vertex set $[n]$)
\end{question}
\begin{answer}
  Intuitively, the answer is immediately $p = 1/2$ since we want the existence of an edge to be as likely as its non existence, generating all graphs at the same probability. Formally, let $X$ denote the number of edges in a given graph $H \in \mathcal{G}(n,p)$. From before, we know that if we generate a random graph, its expected number of edges is $\mathbb{E}_1[X] = \binom{n}{2}p$. In a uniformally random graph, each of the $\binom{n}{2}$ edges is present in exactly half the graphs. So the expected number of edges in a uniformally sampled graph is $\mathbb{E}_2[X] = \frac{\binom{n}{2}}{2}$ so $ p = 1/2$ as the two expectations must be equal. 
\end{answer}

\begin{question}
  What is the expected number of triangles in a random graph $H \in \mathcal{G}(n,p)$
\end{question}
\begin{answer}
  There are $\binom{n}{3}$ unordered triples of vertices. For some fixed triple $\{i, j, k\}$. Let $Y_{ijk}$ be the indicator variable that all 3 edges forming a triangle are present. Since the presence of an individual edge is independent, defined in the natural way. We have that $\mathbb{E}[X_{ijk}] = p^3$. Let $T = \sum_{ijk}Y_{ijk}$ be the number of triangles. Then by linearity of expectation $\mathbb{E}[T] = \binom{n}{3}p^3$.  
\end{answer}


\section{Large Girth and Large Chromatic Number}
\begin{question}
  If $t \geq 2$, what is the expected number of copies of $K_t$ in a random graph $H \in \mathcal{G}(n,p)$?
\end{question}
\begin{answer}
  Let $X_t$ be the number of cliques of order $t$ in $H$. Any given clique has $\binom{t}{2}$ edges so $\mathbb{P}[\text{a given } K_t \text{ is in  } H] = p^{\binom{t}{2}}$. Since any t-tuple can give a clique of order $t$, there are $\binom{n}{t}$ possible ways to choose the vertices for such a clique. Hence by linearity of expectation $\mathbb{E}[X_t] = \binom{n}{t}p^{\binom{t}{2}}$.
\end{answer}
